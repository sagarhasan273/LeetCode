HashMap solution:

class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        map<int, int> mp;
        
        for (auto x: nums){
            mp[x]++;
        }
        vector<pair<int, int>> res;
        vector<int> ans;
        for (auto [n, x]: mp){
            res.push_back({x, n});
        }
        sort(res.rbegin(), res.rend());
        
        for (int i = 0; i < k; i++) ans.push_back(res[i].second);
        
        return ans;
    }
};
