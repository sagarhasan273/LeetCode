class Solution {
public:
    bool isIsomorphic(string s, string t) {
        unordered_map<char, char> m1, m2;
        
        int n = s.size();
        
        for (int i = 0; i < n; i++){
            char a=s[i], b=t[i];
            
            if ((m1.find(a) != m1.end() && m1[a] != b) or (m2.find(b) != m2.end() && m2[b] != a)) return false;
            
            m1[a] = b;
            m2[b] = a;
        }
        
        return true;
    }
};
