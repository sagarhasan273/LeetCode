class Solution:
    def findDisappearedNumbers(self, nums: List[int]) -> List[int]:
        res, l = [], len(nums)
        
        for n in nums:
            i = abs(n) - 1
            nums[i] = -1*abs(nums[i])
        
        res = []
        
        for i in range(l):
            if nums[i] > 0:
                res.append(i+1)
        
        return res
