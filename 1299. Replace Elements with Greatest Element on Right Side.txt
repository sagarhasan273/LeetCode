class Solution {
public:
    vector<int> replaceElements(vector<int>& arr) {
        int l = arr.size()-1;
        int mx = arr[l];
        arr[l] = -1;
        for (int i = l-1; i>=0; i--){
            if (arr[i] > mx){
                int temp = mx;
                mx = arr[i];
                arr[i] = temp;
            }else arr[i] = mx;
        }
        
        return arr;
    }
};
