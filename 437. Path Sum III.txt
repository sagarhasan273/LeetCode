class Solution {
public:
    int ans = 0;
    
    int pathSum(TreeNode* root, int targetSum) {
        if (root){
            pathSum(root->left, targetSum);
            pathSumA(root, targetSum);
            pathSum(root->right, targetSum);
        }
        return ans;
    }
    void pathSumA(TreeNode* root, double target){
        if (!root)return;
        
        if (root->val == target) ans++;
        pathSumA(root->left, target-root->val);
        pathSumA(root->right, target-root->val);
        
    }
};
